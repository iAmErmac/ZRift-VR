ACTOR ChasmPlayer : DoomPlayer
{
  Speed 1
  Health 100
  Radius 16
  Height 56
  Mass 100
  PainChance 255
  Player.StartItem "ChasmPlayerToken"
  //Player.SpawnClass "Fighter"
  Player.DisplayName "Rift Agent"
  Player.StartItem "ChasmRifle"
  Player.StartItem "ChasmFists"
  Player.StartItem "ChasmClip", 50
  Player.SoundClass "ChasmPlayer"
  Player.Portrait "CHASMPL"
  Player.Morphweapon "ChasmChainFist"
  DamageFactor "PlayerSaw", 0.0
  DamageFactor "PlayerEnergy", 0.0
  DamageFactor "ZombiePoison", 0.5
  BloodColor "Red"
  BloodType "RedBlood", "RedBlood"
  States
  {
  Spawn:
    TNT1 A 0 NoDelay
    PLXB ABCDEFGHIJKLM 2
    Loop
  See:
    PLXA ABCDE 2
	TNT1 A 0 A_PlaySound("Player/Step", 7, 0.50)
	PLXA FGHIJ 2
	TNT1 A 0 A_PlaySound("Player/Step", 7, 0.50)
    Goto Spawn
  Missile:
    TNT1 A 0 A_Jump(100, "Missile2")
    PLXC ABCDEF 2
    Goto Spawn
  Missile2:
	PLXD ABCDEF 2
    Goto Spawn
  Melee:
    PLXC ABCDEF 2
    Goto Spawn
  Pain:
    PLXH A 2 A_Pain
    PLXH BCB 2
    Goto Spawn
  Death:
    TNT1 A 0 A_StopSound(CHAN_7)
	TNT1 A 0 A_StopSound(CHAN_6)
	TNT1 A 0 A_StopSound(CHAN_WEAPON)
	TNT1 A 0 A_Jump(50, "Death1")
	TNT1 A 0 A_Jump(50, "Death2")
	TNT1 A 0 A_Jump(50, "Death3")
    PLXE A 2
    PLXE B 2 A_PlayerScream
    PLXE C 2 A_NoBlocking
    PLXE DEFGHIJKLMNOPQRST 2 
    PLXE T -1
    Stop
  Death1:
    PLXF A 2
    PLXF B 2 A_PlayerScream
    PLXF C 2 A_NoBlocking
    PLXF DEFGHIJKLMNOPQRSTUV 2 
    PLXF V -1
    Stop
  Death2:
	PLXG A 2
    PLXG B 2 A_PlayerScream
    PLXG C 2 A_NoBlocking
    PLXG DEFGHIJKLMNOPQRSTUV 2 
    PLXG V -1
    Stop
  Death3:
    PLXH A 2
    PLXH B 2 A_PlayerScream
    PLXH C 2 A_NoBlocking
    PLXH DEFGHIJKLMNOPQRSTUVWXYZ 2 
    PLXH Z -1
    Stop
  }
}
