Class ChasmPlasmaThrower : Weapon
{
	Default
	{
	Inventory.PickupMessage "Picked up a Plasma Thrower.";
	Inventory.PickupSound "Weapon/Pickup";
	Weapon.AmmoType1 "CPlasmaCell";
	Weapon.AmmoType2 "CPlasmaCell";
	Weapon.AmmoUse1 1;
	Weapon.AmmoUse2 5;
	Weapon.AmmoGive 20;
	Weapon.UpSound "PlasmaThrower/Up";
	Weapon.SlotNumber 9;
	Weapon.SelectionOrder 1;
	Weapon.KickBack 1;
	Weapon.BobStyle "Alpha";
	Weapon.BobRangeX 0.07;
	Weapon.BobRangeY 0.45;
	Weapon.BobSpeed 1.90;
	Inventory.Icon "HUPLS1";
	}
	
	States
	{
	Spawn:
		TNT1 A 0 NoDelay;
		PLWX Z -1;
		Stop;
	Ready:
		PLW1 A 1 A_WeaponReady;
		Loop;
	Select:
		PLW1 A 0 SetPlayerProperty(0,1,2);
		PLW1 A 1 A_Raise;
		Wait;
	Deselect:
		PLW1 A 0 A_StopSound(CHAN_WEAPON);
		PLW1 A 1 A_Lower;
		Wait;
	Fire:
		PLW1 A 0 A_PlaySound("PlasmaThrower/Start", 7);
		PLW1 B 1 BRIGHT A_PlaySound("PlasmaThrower/FireLoop", CHAN_WEAPON, 1, 1);
	Hold:

		PLW1 B 0 A_FireProjectile("PlasmaFlame",frandom(-4.6,4.1),1,6,0,0,frandom(-2.1,1.9));
		PLW1 B 0 A_FireProjectile("PlasmaFlame",frandom(-4.6,4.1),0,6,0,0,frandom(-2.1,1.9));
		PLW1 B 0 A_FireProjectile("PlasmaFlame",frandom(-4.6,4.1),0,6,0,0,frandom(-2.1,1.9));
		PLW1 C 2 BRIGHT;
		PLW1 B 1 BRIGHT;
		PLW1 B 0 A_FireProjectile("PlasmaFlame",frandom(-4.6,4.1),1,6,0,0,frandom(-2.1,1.9));
		PLW1 B 0 A_FireProjectile("PlasmaFlame",frandom(-4.6,4.1),0,6,0,0,frandom(-2.1,1.9));
		PLW1 B 0 A_FireProjectile("PlasmaFlame",frandom(-4.6,4.1),0,6,0,0,frandom(-2.1,1.9));
		PLW1 C 2 BRIGHT Offset(-1, 30);
		PLW1 B 1 BRIGHT;
		PLW1 B 0 A_FireProjectile("PlasmaFlame",frandom(-4.6,4.1),1,6,0,0,frandom(-2.1,1.9));
		PLW1 B 0 A_FireProjectile("PlasmaFlame",frandom(-4.6,4.1),0,6,0,0,frandom(-2.1,1.9));
		PLW1 B 0 A_FireProjectile("PlasmaFlame",frandom(-4.6,4.1),0,6,0,0,frandom(-2.1,1.9));
		PLW1 C 2 BRIGHT;
		PLW1 B 1 BRIGHT;
		PLW1 B 0 A_FireProjectile("PlasmaFlame",frandom(-4.6,4.1),1,6,0,0,frandom(-2.1,1.9));
		PLW1 B 0 A_FireProjectile("PlasmaFlame",frandom(-4.6,4.1),0,6,0,0,frandom(-2.1,1.9));
		PLW1 B 0 A_FireProjectile("PlasmaFlame",frandom(-4.6,4.1),0,6,0,0,frandom(-2.1,1.9));
		PLW1 C 2 BRIGHT Offset(1, 34);
		PLW1 B 1 BRIGHT A_ReFire;
	LetGo:
		PLW1 B 0 A_StopSound(CHAN_WEAPON);
		PLW1 B 0 A_PlaySound("PlasmaThrower/Stop", 1);
		PLW1 C 4;
		PLW1 D 4;
		PLW1 E 4;
		PLW1 F 4;
		PLW1 B 4;
		PLW1 A 4;
		Goto Ready;
	AltFire:
		PLW2 A 0;
		PLW2 A 0 A_PlaySound("PlasmaThrower/Charge", 7);
		PLW2 A 0 A_Quake(2, 25, 0, 50, "None");
		PLW2 A 0 BRIGHT A_SetBlend("Blue",0.08,5);
		PLW2 A 16 BRIGHT;
	AltFiring:
		PLW2 B 1 BRIGHT A_PlaySound("PlasmaThrower/Fire", 1);
		PLW2 BBBBBBBB 0 BRIGHT A_FireProjectile("PlasmaPellet",frandom(-11.6,11.1),0,6,0,0,frandom(-1.1,0.9));
		PLW2 BBBBBBBB 0 BRIGHT A_FireProjectile("PlasmaPellet",frandom(11.6,-11.1),0,6,0,0,frandom(-1.1,0.9));
		PLW2 B 0 BRIGHT A_FireProjectile("PlasmaPellet",0,1,6,0,0,frandom(-1.1,0.9));
		PLW2 B 0 A_FireProjectile("PlasmaShortFlame",frandom(-4.6,4.1),0,6,0,0,frandom(-2.1,1.9));
		PLW2 B 0 A_FireProjectile("PlasmaShortFlame",frandom(-4.6,4.1),0,6,0,0,frandom(-2.1,1.9));
		PLW2 B 0 A_FireProjectile("PlasmaShortFlame",frandom(-4.6,4.1),0,6,0,0,frandom(-2.1,1.9));
		PLW2 B 0 A_FireProjectile("PlasmaShortFlame",frandom(-4.6,4.1),0,6,0,0,frandom(-2.1,1.9));
		PLW2 B 0 A_FireProjectile("PlasmaShortFlame",frandom(-4.6,4.1),0,6,0,0,frandom(-2.1,1.9));
		PLW2 B 0 A_FireProjectile("PlasmaShortFlame",frandom(-4.6,4.1),0,6,0,0,frandom(-2.1,1.9));

		PLW2 B 4;
		PLW2 C 3;
		PLW2 C 0 A_PlaySound("PlasmaThrower/Recoil", 3);
