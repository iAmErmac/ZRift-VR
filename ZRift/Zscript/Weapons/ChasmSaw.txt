Class ChasmSaw : Weapon
{
	Default
	{
	Inventory.PickupMessage "Picked up a Saw.";
	Inventory.PickupSound "Ammo/Pickup";
	Weapon.AmmoType1 "SawBlade";
	Weapon.AmmoUse1 1;
	Weapon.AmmoGive 6;
	Weapon.UpSound "Saw/Up";
	Weapon.SlotNumber 8;
	Weapon.SelectionOrder 3;
	Weapon.KickBack 1;
	Weapon.BobStyle "Alpha";
	Weapon.BobRangeX 0.05;
	Weapon.BobRangeY 0.35;
	Weapon.BobSpeed 2;
	Inventory.Icon "HUSAW1";
	+WEAPON.NOALERT;
	}
	
	States
	{
	Spawn:
		TNT1 A 0 NoDelay;
		PSAX A -1;
		Stop;
	Ready:
		DISX Z 1 A_WeaponReady;
		Loop;
	Select:
		TNT1 A 0 SetPlayerProperty(0,1,2);
		DISX CBA 2;
		Goto Ready;
	Deselect:
		DISX Z 1 A_Lower;
		Loop;
	Fire:
		TNT1 A 0 A_Jump(100, "Fire2");
		DISX ABC 1;
		TNT1 A 0 A_PlaySound("ChasmSaw/Fire",1);
		DISX C 0 BRIGHT A_FireCustomMissile("ChasmSawBlade");
		TNT1 A 2;
		DISX DA 1;
		DISX A 1;
		TNT1 A 0 BRIGHT A_ReFire;
		Goto Ready;
	Fire2:
		DISX EFG 1;
		TNT1 A 0 A_PlaySound("ChasmSaw/Fire",1);
		DISX G 0 BRIGHT A_FireCustomMissile("ChasmSawBlade");
		TNT1 A 2;
		DISX HA 1;
		DISX A 1;
		TNT1 A 0 BRIGHT A_ReFire;
		Goto Ready;
	}
}
